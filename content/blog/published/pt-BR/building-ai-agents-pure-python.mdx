O primeiro princ√≠pio de constru√ß√£o de agentes eficazes √© estabelecer comunica√ß√£o direta e confi√°vel com APIs de LLM. Isso significa entender como estruturar requests, gerenciar responses e lidar com erros de forma elegante.

```python
import openai
import json
from typing import Dict, List, Optional, Any
from dataclasses import dataclass
from enum import Enum

class MessageRole(Enum):
    SYSTEM = "system"
    USER = "user"
    ASSISTANT = "assistant"

@dataclass
class Message:
    role: MessageRole
    content: str
    
    def to_dict(self) -> Dict[str, str]:
        return {
            "role": self.role.value,
            "content": self.content
        }

class OpenAIClient:
    def __init__(self, api_key: str, model: str = "gpt-4"):
        self.client = openai.OpenAI(api_key=api_key)
        self.model = model
    
    def create_completion(
        self, 
        messages: List[Message],
        temperature: float = 0.7,
        max_tokens: Optional[int] = None
    ) -> str:
        """
        Cria uma completion usando a API OpenAI com tratamento de erro robusto.
        """
        try:
            response = self.client.chat.completions.create(
                model=self.model,
                messages=[msg.to_dict() for msg in messages],
                temperature=temperature,
                max_tokens=max_tokens
            )
            return response.choices[0].message.content
        except Exception as e:
            raise Exception(f"Erro na API: {str(e)}")
```

```python
from abc import ABC, abstractmethod
from typing import Any, Dict, List, Optional
import logging

class BaseAgent(ABC):
    """
    Classe base abstrata para todos os agentes de IA.
    Define a interface e comportamento comum.
    """
    
    def __init__(self, name: str, openai_client: OpenAIClient):
        self.name = name
        self.client = openai_client
        self.conversation_history: List[Message] = []
        self.logger = logging.getLogger(f"Agent.{name}")
    
    @abstractmethod
    def get_system_prompt(self) -> str:
        """
        Define o prompt do sistema que estabelece a personalidade e comportamento do agente.
        """
        pass
    
    @abstractmethod
    def process_input(self, user_input: str) -> str:
        """
        Processa entrada do usu√°rio e retorna resposta.
        """
        pass
    
    def add_system_message(self) -> None:
        """
        Adiciona mensagem do sistema ao hist√≥rico da conversa.
        """
        system_msg = Message(MessageRole.SYSTEM, self.get_system_prompt())
        self.conversation_history.insert(0, system_msg)
    
    def add_user_message(self, content: str) -> None:
        """
        Adiciona mensagem do usu√°rio ao hist√≥rico.
        """
        user_msg = Message(MessageRole.USER, content)
        self.conversation_history.append(user_msg)
    
    def add_assistant_message(self, content: str) -> None:
        """
        Adiciona resposta do assistente ao hist√≥rico.
        """
        assistant_msg = Message(MessageRole.ASSISTANT, content)
        self.conversation_history.append(assistant_msg)
    
    def clear_history(self) -> None:
        """
        Limpa o hist√≥rico da conversa, mantendo apenas a mensagem do sistema.
        """
        system_messages = [msg for msg in self.conversation_history 
                          if msg.role == MessageRole.SYSTEM]
        self.conversation_history = system_messages
    
    def get_conversation_context(self, max_messages: int = 20) -> List[Message]:
        """
        Retorna contexto limitado da conversa para evitar exceder limites de token.
        """
        if len(self.conversation_history) <= max_messages:
            return self.conversation_history
        
        # Sempre manter mensagem do sistema + mensagens mais recentes
        system_messages = [msg for msg in self.conversation_history 
                          if msg.role == MessageRole.SYSTEM]
        recent_messages = self.conversation_history[-(max_messages-len(system_messages)):]
        
        return system_messages + recent_messages
```

```python
class MultiAgentWorkflow:
    """
    Coordena m√∫ltiplos agentes para fluxos de trabalho complexos.
    """
    
    def __init__(self, openai_client: OpenAIClient):
        self.client = openai_client
        self.agents: Dict[str, BaseAgent] = {}
        self.workflow_history: List[Dict[str, Any]] = []
    
    def register_agent(self, agent: BaseAgent) -> None:
        """
        Registra um agente no workflow.
        """
        self.agents[agent.name] = agent
    
    def execute_sequential_workflow(self, tasks: List[Dict[str, Any]]) -> List[str]:
        """
        Executa tarefas sequencialmente atrav√©s de diferentes agentes.
        """
        results = []
        
        for task in tasks:
            agent_name = task.get("agent")
            input_data = task.get("input")
            
            if agent_name not in self.agents:
                raise ValueError(f"Agente '{agent_name}' n√£o encontrado")
            
            agent = self.agents[agent_name]
            result = agent.process_input(input_data)
            results.append(result)
            
            # Log do workflow
            self.workflow_history.append({
                "agent": agent_name,
                "input": input_data,
                "output": result,
                "timestamp": "2024-12-01T10:00:00Z"
            })
        
        return results
    
    def execute_parallel_analysis(self, input_data: str, agent_names: List[str]) -> Dict[str, str]:
        """
        Executa an√°lise paralela com m√∫ltiplos agentes.
        """
        results = {}
        
        for agent_name in agent_names:
            if agent_name in self.agents:
                results[agent_name] = self.agents[agent_name].process_input(input_data)
        
        return results
```

### 2. Padr√£o de Ferramentas (Tool Pattern)

```python
from abc import ABC, abstractmethod

class Tool(ABC):
    """
    Classe base para ferramentas que agentes podem usar.
    """
    
    @abstractmethod
    def get_name(self) -> str:
        pass
    
    @abstractmethod
    def get_description(self) -> str:
        pass
    
    @abstractmethod
    def execute(self, **kwargs) -> Any:
        pass

class FileSystemTool(Tool):
    """
    Ferramenta para opera√ß√µes do sistema de arquivos.
    """
    
    def get_name(self) -> str:
        return "file_system"
    
    def get_description(self) -> str:
        return "L√™, escreve e lista arquivos no sistema de arquivos"
    
    def execute(self, operation: str, path: str, content: str = None) -> Any:
        """
        Executa opera√ß√µes do sistema de arquivos de forma segura.
        """
        import os
        
        # Valida√ß√£o de seguran√ßa b√°sica
        if ".." in path or path.startswith("/"):
            raise ValueError("Caminho de arquivo inseguro")
        
        if operation == "read":
            try:
                with open(path, 'r', encoding='utf-8') as f:
                    return f.read()
            except FileNotFoundError:
                return f"Arquivo n√£o encontrado: {path}"
        
        elif operation == "write":
            if content is None:
                raise ValueError("Conte√∫do √© obrigat√≥rio para opera√ß√£o de escrita")
            
            with open(path, 'w', encoding='utf-8') as f:
                f.write(content)
            return f"Arquivo escrito com sucesso: {path}"
        
        elif operation == "list":
            if os.path.isdir(path):
                return os.listdir(path)
            else:
                return []
        
        else:
            raise ValueError(f"Opera√ß√£o n√£o suportada: {operation}")

class ToolEnabledAgent(BaseAgent):
    """
    Agente que pode usar ferramentas externas.
    """
    
    def __init__(self, name: str, openai_client: OpenAIClient):
        super().__init__(name, openai_client)
        self.tools: Dict[str, Tool] = {}
    
    def register_tool(self, tool: Tool) -> None:
        """
        Registra uma ferramenta para uso pelo agente.
        """
        self.tools[tool.get_name()] = tool
    
    def get_tools_description(self) -> str:
        """
        Retorna descri√ß√£o de todas as ferramentas dispon√≠veis.
        """
        if not self.tools:
            return "Nenhuma ferramenta dispon√≠vel."
        
        descriptions = []
        for tool in self.tools.values():
            descriptions.append(f"- {tool.get_name()}: {tool.get_description()}")
        
        return "Ferramentas dispon√≠veis:\n" + "\n".join(descriptions)
    
    def use_tool(self, tool_name: str, **kwargs) -> Any:
        """
        Usa uma ferramenta espec√≠fica.
        """
        if tool_name not in self.tools:
            raise ValueError(f"Ferramenta '{tool_name}' n√£o encontrada")
        
        tool = self.tools[tool_name]
        return tool.execute(**kwargs)
```

### 2. Sistema de M√©tricas

```python
from collections import defaultdict, Counter
import time
from typing import List, Dict, Any

class AgentMetrics:
    """
    Coleta e analisa m√©tricas de performance de agentes.
    """
    
    def __init__(self):
        self.interaction_count = 0
        self.error_count = 0
        self.response_times: List[float] = []
        self.error_types = Counter()
        self.daily_interactions = defaultdict(int)
    
    def record_interaction(self, response_time: float) -> None:
        """
        Registra uma intera√ß√£o bem-sucedida.
        """
        self.interaction_count += 1
        self.response_times.append(response_time)
        
        # Registrar por dia
        today = datetime.utcnow().date().isoformat()
        self.daily_interactions[today] += 1
    
    def record_error(self, error_type: str) -> None:
        """
        Registra um erro.
        """
        self.error_count += 1
        self.error_types[error_type] += 1
    
    def get_metrics_summary(self) -> Dict[str, Any]:
        """
        Retorna resumo das m√©tricas.
        """
        avg_response_time = (
            sum(self.response_times) / len(self.response_times) 
            if self.response_times else 0
        )
        
        success_rate = (
            (self.interaction_count / (self.interaction_count + self.error_count)) * 100
            if (self.interaction_count + self.error_count) > 0 else 0
        )
        
        return {
            "total_interactions": self.interaction_count,
            "total_errors": self.error_count,
            "success_rate_percent": round(success_rate, 2),
            "average_response_time_seconds": round(avg_response_time, 3),
            "most_common_errors": dict(self.error_types.most_common(5)),
            "daily_interactions": dict(self.daily_interactions)
        }
```
